function loadHyperlinks() { fetch('https://raw.githubusercontent.com/JC-DMD/QuickLinks/main/HyperLinks.txt') .then(response => response.text()) .then(data => { let lines = data.split('\n'); let list = document.getElementById('linksList'); lines.forEach(line => { if (line.trim() === '') return; let parts = line.match(/\[(.*?)\]\((.*?)\)/); if (parts && parts.length === 3) { let text = parts[1]; let url = parts[2]; let listItem = document.createElement('li'); let link = document.createElement('a'); let button = document.createElement('button'); link.href = url; link.textContent = text; link.target = "_blank"; // Open in new tab button.textContent = 'Copy'; button.onclick = function() { navigator.clipboard.writeText('[' + text + '](' + url + ')'); }; listItem.appendChild(button); listItem.appendChild(link); list.appendChild(listItem); } }); }) .catch(error => console.error('Error:', error)); } // Call the function when the window loads window.onload = loadHyperlinks;
